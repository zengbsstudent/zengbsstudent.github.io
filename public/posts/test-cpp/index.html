#include <iostream>
#include <vector>

// 這是 C++ 的關鍵字上色測試
int main() {
    std::vector<std::string> msg {"Hello", "Hugo", "Chroma"};

    for (const std::string& word : msg) {
        std::cout << word << " ";
    }
    std::cout << std::endl;

    int myNumber = 42;
    if (myNumber > 40) {
        return 0;
    }

    return 1;
}
